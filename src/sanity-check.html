Method 3: Python Script

#!/usr/bin/env python3
import PyPDF2
import re

with open("Rulebook INSTRUO'14.pdf", 'rb') as file:
    pdf_reader = PyPDF2.PdfReader(file)
    for page in pdf_reader.pages:
        text = page.extract_text()
        if "EOF{" in text:
            # Extract and print flag
            flags = re.findall(r'EOF\{[^}]+\}', text)
            for flag in flags:
                print(flag)
